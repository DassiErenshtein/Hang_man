# 注驻转 驻拽爪 拽转 专 砖转砖 爪注转 注转
def check_is_login(func):
    @wraps(func)  # 砖专 -转 砖 驻拽爪 拽专转
    def wrapper(*args, **kwargs):
        # 拽转 拽 注 砖 'user_id'
        if request.cookies.get('user_id') == None:
            return abort(403)  # 砖转砖  专 - 砖转 HTTP 403
        return func(*args, **kwargs)  # 拽专 驻拽爪  砖转砖 专

    return wrapper


# 驻拽爪 拽转 转拽转 转注转 转
def check_id(id):
    if len(id) != 9:  # 专 砖
        return False
    sum = 0  # 住 住驻专转 砖
    i1 = 1  # 砖拽 砖 住驻专
    check = int(id[-1])  # 住驻专转 拽专转
    id = id[0:-1]  # 住专转 住驻专转 拽专转
    for i in id:
        try:
            int(i)
        except ValueError:
            return False
        num = i1 * int(i)  # 驻 砖拽
        while num >= 1:  # 砖 住驻专转 住驻专
            sum += num % 10
            num -= num % 10
            num /= 10

        i1 = 1 if i1 == 2 else 2  # 砖 砖拽
    # 拽  住驻专转 拽专转 转转
    return check == 10 - sum % 10 or check == 0 == sum % 10

# 驻拽爪 拽  砖拽 拽 驻 转注转 转
def exists_player(id):
    if path.exists(path_file_name):# 拽  拽抓 砖转砖 拽
        with open(path_file_name, 'r', encoding="UTF-8") as f:
            list1 = f.readlines()
            for l in list1:
                l1 = l.removesuffix('\n')
                player = literal_eval(l1)# 专转 专转 拽
                if player['id'] == id:# 拽转 转 转注转 转
                    return player
    return False


@cross_origin(app, supports_credentials=True)
@app.route('/login', methods=["GET"])
def login():
    user = request.json# 拽专转 转 砖转砖
    res = check_id(user['id'])# 拽转 转拽转 转注转 转
    if (res == False):
        return abort(400)
    if path.exists(path_file_name): # 拽  拽抓 砖转砖 拽
        with open(path_file_name, 'r', encoding="UTF-8") as f:
            list1 = f.readlines()
            for l in list1:
                if l != '\n':
                    l1 = l.removesuffix('\n')
                    try:
                        player = literal_eval(l1)# 专转 专转 拽
                    except:
                        return abort(500)
                    # 转 砖 砖转砖 住住
                    if player['id'] == user['id'] and player['password'] == user['password']:
                        response = make_response(jsonify({"message": f"Hello {player['name']}!"}))
                        # 专转 注转 注 转 砖转砖
                        response.set_cookie('user_name', player['name'], max_age=600, httponly=True)
                        response.set_cookie('user_id', player['id'], max_age=600, httponly=True)
                        return response
    return abort(405)# 砖转砖  爪
